int	check_base(char *base)
{
	int	i;
	int	j;

	if (base[0] == '\0' && base[1] == '\0')
		return (0);
	i = 0;
	while(base[i] != '\0')
	{
		if (base[i] == '-' || base[i] == '+')
			return (0);
		j = i + 1;
		while(base[j] != '\0')
		{
			if (base[i] == base[j])
				return (0);
			j++;
		}
		i++;
	}
	return (1);
}
